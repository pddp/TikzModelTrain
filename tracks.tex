\tikzstyle{track}=[
    postaction={draw=gray,densely dashed,line width=14pt},
    postaction={draw,decorate,decoration={curveto,raise=4pt},line width=2pt},
    postaction={draw,decorate,decoration={curveto,raise=-4pt},line width=2pt}]
